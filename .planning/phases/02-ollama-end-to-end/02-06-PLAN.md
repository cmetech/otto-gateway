    ```
    // decodeJSONBody applies http.MaxBytesReader(max) to r.Body then JSON-decodes into dst.
    // EVERY adapter handler that reads a request body MUST use this helper so the body
    // size cap is uniformly enforced (Codex M-5 — was previously only on chat/generate).
    func decodeJSONBody[T any](w http.ResponseWriter, r *http.Request, max int64, dst *T) error {
        r.Body = http.MaxBytesReader(w, r.Body, max)
        dec := json.NewDecoder(r.Body)
        dec.DisallowUnknownFields()  // optional — flip to allow if LangFlow sends extra fields
        if err := dec.Decode(dst); err != nil {
            var maxErr *http.MaxBytesError
            if errors.As(err, &maxErr) {
                return fmt.Errorf("decode: request body exceeds %d bytes: %w", max, err)
            }
            return fmt.Errorf("decode: %w", err)
        }
        return nil
    ```

Plan 06 mounts the HTTP server. Every inbound request crosses the network →
